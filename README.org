- Changes ::

  2. ~CMD~ directive: use proper JSON notation
  3. Fixed typo in ~RUN~ directive
  4. ~EXPOSE~ app port
  5. Pin image and package tags to specific versions

- Updated dockerfile ::

RUN apt-get update && \
    apt-get install -y nodejs=18.13.0+dfsg1-1

WORKDIR /usr/src/app

COPY package*.json index.js ./

EXPOSE 3000

CMD ["node", "index.js"]
#+end_src
